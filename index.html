CaseTrigger.cls + CaseAssignmentEngine.cls
Bulkified
// Thin trigger — 100% logic in Queueable handler
trigger CaseTrigger on Case (after insert) {
    System.enqueueJob(new CaseAssignmentEngine(Trigger.newMap.keySet()));
}

public class CaseAssignmentEngine implements Queueable {
    private Set<Id> caseIds;
    public CaseAssignmentEngine(Set<Id> ids) { this.caseIds = ids; }

    public void execute(QueueableContext ctx) {
        // Single SOQL — bulkified across all inserted cases
        List<Case> cases = [SELECT Id, Priority, Type, OwnerId
                             FROM Case WHERE Id IN :caseIds AND Status = 'New'];
        Map<String, Id> queueMap = getQueueMap();
        List<Case> toUpdate = new List<Case>();

        for (Case c : cases) {
            String q = resolveQueue(c.Priority, c.Type);
            if (queueMap.containsKey(q)) { c.OwnerId = queueMap.get(q); toUpdate.add(c); }
        }
        // Partial success — errors logged, batch not aborted
        if (!toUpdate.isEmpty()) logErrors(Database.update(toUpdate, false));
    }
}

leaveRequestForm.js
LWC
import { LightningElement, track, wire } from 'lwc';
import { ShowToastEvent } from 'lightning/platformShowToastEvent';
import getLeaveBalance from '@salesforce/apex/LeaveController.getLeaveBalance';
import submitLeaveRequest from '@salesforce/apex/LeaveController.submitLeaveRequest';

export default class LeaveRequestForm extends LightningElement {
    @track leaveType = '';
    @track availableBalance = 0;
    @track isSubmitting = false;

    // Reactive: wire re-fetches when leaveType changes
    @wire(getLeaveBalance, { leaveType: '$leaveType' })
    wiredBalance({ data, error }) {
        if (data) this.availableBalance = data;
    }

    handleSubmit() {
        const days = this.calcDays(this.startDate, this.endDate);
        if (days > this.availableBalance) {
            this.dispatchEvent(new ShowToastEvent({
                title: 'Insufficient Balance',
                message: `Only ${this.availableBalance} day(s) available.`,
                variant: 'warning'
            }));
            return;
        }
        this.isSubmitting = true;
        submitLeaveRequest({ leaveType: this.leaveType, days })
            .then(() => this.dispatchEvent(new ShowToastEvent({
                title: 'Success', message: 'Submitted for approval.', variant: 'success'
            })))
            .finally(() => this.isSubmitting = false);
    }
}

ERPIntegrationService.cls
REST API
public class ERPIntegrationService {

    @future(callout=true)  // async — avoids "callout after DML" exception
    public static void syncAccountToERP(Id accountId) {
        Account acc = [SELECT Id, Name, Phone, BillingCity, Industry
                       FROM Account WHERE Id = :accountId LIMIT 1];

        // Wrapper class builds clean JSON — no string concat risk
        HttpRequest req = new HttpRequest();
        req.setEndpoint('callout:ERP_System/api/accounts'); // Named Credential
        req.setMethod('POST');
        req.setHeader('Content-Type', 'application/json');
        req.setBody(JSON.serialize(new ERPPayload(acc)));
        req.setTimeout(10000);

        HttpResponse res = new Http().send(req);
        if (res.getStatusCode() != 200)
            logIntegrationError(accountId, res.getStatusCode(), res.getBody());
    }

    public class ERPPayload {
        public String sfId, name, phone, city, industry;
        public ERPPayload(Account a) {
            sfId=a.Id; name=a.Name; phone=a.Phone; city=a.BillingCity; industry=a.Industry;
        }
    }
}

How I Think as a Salesforce Developer
🎯
Requirements before code
Map business process first — objects, data volume, user roles. Design for 10,000 records from day one, not 10.
⚖️
Declarative-first principle
If a Flow solves it cleanly — use it. Apex is a deliberate choice for bulk operations or complex logic, not the default.
📏
Governor limits by design
Plan around 150 SOQL / 10K DML / 100 callout limits before writing line one. Async Apex is first-class, not a hotfix.
🔒
Security model is non-negotiable
CRUD, FLS, and sharing rules by design. Deliberate with/without sharing decisions in every Apex class.
🔍
Debug before guess
Developer Console logs, structured System.debug, test class isolation — understand root cause before adding workarounds.
🔄
Maintenance is a feature
Thin triggers, named constants, test methods describing behavior not coverage. Code is read 10× more than it's written.
